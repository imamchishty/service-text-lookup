1   package com.demo.service.text.lookup.rest.controller;
2   
3   import com.demo.service.text.lookup.exception.GlobalBusinessCodes;
4   import com.demo.service.text.lookup.model.Text;
5   import com.demo.service.text.lookup.rest.constant.ApiConstants;
6   import com.shedhack.exception.core.BusinessException;
7   import org.springframework.beans.factory.annotation.Autowired;
8   import org.springframework.data.redis.core.StringRedisTemplate;
9   import org.springframework.web.bind.annotation.*;
10  
11  /**
12   * <pre>
13   *     Text Controller
14   * </pre>
15   */
16  @RestController
17  public class TextLookupController {
18  
19      @Autowired
20      private StringRedisTemplate template;
21  
22      /**
23       * Attempts to return the 'String' value based on the key.
24       * @param key primary key for the redis store
25       * @return the value stored in redis
26       * @throws BusinessException if the key is not found
27       */
28      @RequestMapping(path = ApiConstants.API_TEXT_LOOKUP, method = RequestMethod.GET)
29      public Text lookupTextByKey(@PathVariable String key) {
30  
31          // Attempt to get the key from Redis
32          String value = template.opsForValue().get(key);
33  
34          // No key, throw exception
35          if( value == null || value.length() == 0) {
36  
37              throw BusinessException.builder("Unable to find key").withParam("key", key)
38                      .generateId().withBusinessCode(GlobalBusinessCodes.TLS001)
39                      .build();
40          }
41  
42          return new Text(key, value);
43      }
44  
45      @RequestMapping(path = ApiConstants.API_TEXT_ROOT, method = RequestMethod.POST)
46      public Text storeKeyValue(@RequestBody Text text) {
47  
48          // Attempt to save the key/value combo
49          template.opsForValue().set(text.getKey(), text.getValue());
50  
51          return text;
52      }
53  
54  
55  }
